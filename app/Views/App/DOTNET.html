.NET - Notifier Mobile Service

Download

You can download Notifier Mobile Service as a DLL file.

Download

Or

To install Notifier Mobile Service, run the following command in the Package Manager Console

PM> Install-Package NotifierMobileService

By Notifier Mobile Service, you can CREATE/GET/UPDATE/DELETE notifications easily just with a few lines of code.

Import library

    using NotifierMobile;

Create Notification

    // Create object for authentication
    Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

    // Create new notification
    Notification notification = new Notification();
    notification.Title = "title";
    notification.Message = "message";
    notification.Type = (int) NotificationType.INFO;

    // Add notification by service
    NotifierMobileService.Add(notification, authentication);

Note: Notification id is generated automatically, you do not need to set id for new notification

Get All Notifications

    // Create object for authentication
    Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

    // Get all notifications by service
    NotifierMobileService.GetAll(authentication);

Note: You can get notifications by type (INFO, WARNING, ERROR) or status of notification (unread or read)

Get Notification By Id

    // Create object for authentication
    Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

    // Get notification with id = 1 by service
    NotifierMobileService.Get(1, authentication);

Update Notification

    // Create object for authentication
    Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

    Notification notification = new Notification();
    // Update message
    notification.Message = "message";

    // Update message for notification with id = 1 by service
    NotifierMobileService.Update(1, notification, authentication);

Note: You can not update id of notification

Mark Notification As Read

    // Create object for authentication
    Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

    // Mark notification with id = 1 as read by service
    NotifierMobileService.MarkAsRead(1, authentication);

Delete Notification

    // Create object for authentication
    Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

    // Delete notification with id = 1 by service
    NotifierMobileService.Delete(1, authentication);

Exception

A NotificationException will be throwed when there is an error during connecting to server.

    try
    {
        // Create object for authentication
        Authentication authentication = new Authentication("yourUsername", "yourSecretKey");

        // Create new notification
        Notification notification = new Notification();
        notification.Title = "title";
        notification.Message = "message";
        notification.Type = (int) NotificationType.INFO;

        // Add notification by service
        NotifierMobileService.Add(notification, authentication);
    }
    catch (NotificationExcception e)
    {
        Console.WriteLine(e.ToString());
    }
